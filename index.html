async function refresh() { try { const res = await fetch('/list-servers', { headers: { 'x-api-key': apiKey } }); if(res.status === 401) { location.reload(); return; } const data = await res.json(); masterData = data; // Store latest data const grid = document.getElementById('server-grid'); // We clear the grid entirely before rebuilding to prevent visual ghosting grid.innerHTML = ''; // Sort IDs so they always appear in the same order (prevents jumping) const sortedIds = Object.keys(data.servers).sort(); if (sortedIds.length === 0) { grid.innerHTML = '
No active instances found.
'; } for (const id of sortedIds) { const s = data.servers[id]; const div = document.createElement('div'); div.className = 'card'; div.id = 'card-' + id; div.innerHTML = `
${s.isStudio ? 'STUDIO' : 'LIVE'}
${s.playerCount} Users
ID: ${id}
`; grid.appendChild(div); } // Update Global Chat const chatBox = document.getElementById('global-chat'); chatBox.innerHTML = data.global.map(c => `
${c.jobId.substring(0,6)} ${c.t}
${c.msg}
`).join(''); } catch (err) { console.error("Fetch error:", err); } }
